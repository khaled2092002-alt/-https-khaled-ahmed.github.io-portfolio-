// ===== Google Apps Script - Email Web App ===== // 1. روح https://script.google.com // 2. اعمل مشروع جديد // 3. الصق الكود ده كله // 4. غير TO_EMAIL لإيميلك // 5. ملف > حفظ // 6. نشر > نشر كتطبيق ويب // 7. صلاحية: Anyone (حتى مجهول) // 8. انسخ الرابط وحطه في HTML const TO_EMAIL = "khaled2092002@gmail.com"; function doPost(e) { try { const data = e.parameter; const name = data.name || "بدون اسم"; const email = data.email || "بدون إيميل"; const subject = data.subject || "بدون موضوع"; const message = data.message || "بدون رسالة"; const body = ` الاسم: ${name} البريد الإلكتروني: ${email} الموضوع: ${subject} الرسالة: ${message} `; MailApp.sendEmail({ to: TO_EMAIL, subject: `رسالة جديدة من موقع خالد أحمد - ${subject}`, body: body, replyTo: email }); return ContentService .createTextOutput(JSON.stringify({ success: true })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet() { return HtmlService.createHtmlOutput("

Email Web App is running.
